Project 2: Text-Based Game

This is a simple text-based adventure game developed in Java where you can embark on quests, battle enemies, and level up. Below is the code:


                import java.util.Random;
                import java.util.Scanner;
                
                interface Character {
                    void showStats();
                }
                
                interface Quest {
                    void playQuest(Player player);
                }
                
                class Player implements Character {
                    String name;
                    int health = 100;
                    int coins = 50;
                    int experience = 0;
                    int level = 1;
                
                    Player(String name) {
                        this.name = name;
                    }
                
                    void gainCoins(int amount) {
                        coins += amount;
                        System.out.println("You earned " + amount + " coins! Total coins: " + coins);
                    }
                
                    void gainExperience(int amount) {
                        experience += amount;
                        if (experience >= level * 100) {
                            levelUp();
                        }
                    }
                
                    void levelUp() {
                        level++;
                        health += 20;
                        System.out.println("Congratulations! You've leveled up to Level " + level + "!");
                    }
                
                    @Override
                    public void showStats() {
                        System.out.println("\n--- Player Stats ---");
                        System.out.println("Name: " + name);
                        System.out.println("Health: " + health);
                        System.out.println("Coins: " + coins);
                        System.out.println("Level: " + level);
                        System.out.println("Experience: " + experience);
                        System.out.println("--------------------");
                    }
                }
                
                class Enemy {
                    String name;
                    int health;
                    int damage;
                
                    Enemy(String name, int health, int damage) {
                        this.name = name;
                        this.health = health;
                        this.damage = damage;
                    }
                
                    void attack(Player player) {
                        System.out.println(name + " attacks you for " + damage + " damage!");
                        player.health -= damage;
                    }
                
                    void showStats() {
                        System.out.println(name + " - Health: " + health + " | Damage: " + damage);
                    }
                }
                
                class Quest1 implements Quest {
                    @Override
                    public void playQuest(Player player) {
                        System.out.println("\n--- Quest: The Goblin Encounter ---");
                        Random rand = new Random();
                        Enemy goblin = new Enemy("Goblin", 50, 10);
                        goblin.showStats();
                
                        Scanner scanner = new Scanner(System.in);
                        while (goblin.health > 0 && player.health > 0) {
                            System.out.println("\nDo you want to attack or flee?");
                            System.out.println("1. Attack");
                            System.out.println("2. Flee");
                            int choice = scanner.nextInt();
                
                            if (choice == 1) {
                                int damage = rand.nextInt(15) + 5;
                                System.out.println("You attack the Goblin for " + damage + " damage!");
                                goblin.health -= damage;
                                player.gainExperience(20);
                                System.out.println("You gained 20 experience points.");
                            } else if (choice == 2) {
                                System.out.println("You flee the battle.");
                                break;
                            }
                
                            if (goblin.health > 0) {
                                goblin.attack(player);
                            }
                        }
                
                        if (player.health > 0) {
                            System.out.println("\nYou defeated the Goblin and earned 50 coins!");
                            player.gainCoins(50);
                        } else {
                            System.out.println("\nYou were defeated by the Goblin. Game Over.");
                        }
                    }
                }
                
                public class TextBasedGame {
                    public static void main(String[] args) {
                        Scanner scanner = new Scanner(System.in);
                
                        System.out.println("Welcome to the Text-Based Adventure Game!");
                        System.out.print("Enter your character's name: ");
                        String name = scanner.nextLine();
                        Player player = new Player(name);
                
                        player.showStats();
                        
                        Quest quest1 = new Quest1();
                        quest1.playQuest(player);
                
                        player.showStats();
                    }
                }
